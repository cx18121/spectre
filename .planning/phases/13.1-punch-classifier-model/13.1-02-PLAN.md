**normalize_window(kps)** — CRITICAL: this must be bit-for-bit identical to `normalizeWindow()` in `fps/src/lib/normalizeWindow.ts`:
```python
def normalize_window(kps: np.ndarray) -> np.ndarray:
    """
    kps: shape (T, J, 3) — T frames, 8 joints in JOINT_ORDER, xyz in meters
    Returns: same shape, translated to shoulder midpoint at origin, scaled by shoulder width.
    Joints at index 0=LEFT_SHOULDER, 1=RIGHT_SHOULDER in the extracted subset.
    """
    left_sh = kps[:, 0, :]   # (T, 3)
    right_sh = kps[:, 1, :]  # (T, 3)
    midpoint = (left_sh + right_sh) / 2.0   # (T, 3)
    sw = np.linalg.norm(right_sh - left_sh, axis=1, keepdims=True)  # (T, 1)
    sw = np.clip(sw, 1e-6, None)
    normalized = (kps - midpoint[:, np.newaxis, :]) / sw[:, np.newaxis, :]
    return normalized
```

**flip_sequence(kps, label)** — horizontal flip augmentation (per research A4):
```python
SWAP_PAIRS = [(0, 1), (2, 3), (4, 5), (6, 7)]
LABEL_SWAP = {'jab': 'cross', 'cross': 'jab', 'hook_l': 'hook_r', 'hook_r': 'hook_l', 'guard': 'guard'}
def flip_sequence(kps, label):
    flipped = kps.copy()
    for i, j in SWAP_PAIRS:
        flipped[:, i, :], flipped[:, j, :] = kps[:, j, :].copy(), kps[:, i, :].copy()
    flipped[:, :, 0] *= -1
    return flipped, LABEL_SWAP[label]
```

**PunchMLP** — exact architecture from research B1:
```python
class PunchMLP(nn.Module):
    def __init__(self, input_dim=480, num_classes=5):  # 480 = 20*8*3
        super().__init__()
        self.net = nn.Sequential(
            nn.Linear(input_dim, 256), nn.ReLU(), nn.Dropout(0.3),
            nn.Linear(256, 128), nn.ReLU(), nn.Dropout(0.3),
            nn.Linear(128, 64), nn.ReLU(),
            nn.Linear(64, num_classes),
        )
    def forward(self, x):
        return self.net(x.view(x.size(0), -1))
```

```python
import onnx
from onnx import helper, TensorProto

# Minimal valid ONNX: identity model with correct input/output names and shape
# This is a PLACEHOLDER — replace with real model after training
X = helper.make_tensor_value_info('input', TensorProto.FLOAT, [1, 20, 8, 3])
Y = helper.make_tensor_value_info('logits', TensorProto.FLOAT, [1, 5])

# Single Flatten + MatMul + bias node as minimal valid graph
# Actually use a Reshape + Gemm so it's a real executable model
flatten_out = helper.make_tensor_value_info('flat', TensorProto.FLOAT, [1, 480])
reshape_node = helper.make_node('Reshape', inputs=['input', 'shape'], outputs=['flat'])
# Shape initializer
shape_init = onnx.numpy_helper.from_array(
    np.array([1, 480], dtype=np.int64), name='shape'
)
# Weight and bias for a 480->5 linear layer (all zeros = uniform output)
import numpy as np
W_np = np.zeros((5, 480), dtype=np.float32)
b_np = np.zeros((5,), dtype=np.float32)
W_init = onnx.numpy_helper.from_array(W_np, name='W')
b_init = onnx.numpy_helper.from_array(b_np, name='b')
gemm_node = helper.make_node('Gemm', inputs=['flat', 'W', 'b'], outputs=['logits'],
                              transB=1)
graph = helper.make_graph([reshape_node, gemm_node], 'placeholder',
                           [X], [Y], initializer=[shape_init, W_init, b_init])
model = helper.make_model(graph, opset_imports=[helper.make_opsetid('', 17)])
model.ir_version = 8
onnx.save(model, 'fps/public/models/punch_classifier_int8.onnx')
```
